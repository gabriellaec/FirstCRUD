SET @@global.time_zone = '+3:00';
CREATE DATABASE projetoum;
USE projetoum;
CREATE TABLE Tarefas(id INT NOT NULL AUTO_INCREMENT PRIMARY KEY, 
nome_tarefa VARCHAR(100), conteudo VARCHAR(255), data TIMESTAMP NOT NULL, categoria VARCHAR(32) NOT NULL, nome_usuario VARCHAR(32));
CREATE TABLE Usuario(id INT NOT NULL AUTO_INCREMENT PRIMARY KEY, nome VARCHAR(32) NOT NULL, senha  VARCHAR(32) NOT NULL);
